```python
    Posizione originale (1.5, 0.0, π=W) era unfair: heading W spingeva il robot
    in muro centrale entro 60 step. Nuova posizione (3.5, -0.5, π/2=N) ha

    Vedi analysis/maze2_geom_check.py per analisi geometrica completa.

    import math
    spawns_2 = SPAWN_LISTS[2]
    # Cerca la voce D1 (qualsiasi voce con x in [3.0, 4.0] e y in [-1.0, 0.0])
    d_candidates = [s for s in spawns_2 if 3.0 <= s[0] <= 4.0 and -1.0 <= s[1] <= 0.0]

        f"Atteso esattamente 1 spawn in zona D centro-destra, trovati {len(d_candidates)}: {d_candidates}"
```

```python
    def step_action(self, action_index: int):
        cmd = Twist()
        cmd.linear.x  = LINEAR_VEL
        cmd.angular.z = -0.8 + 0.16 * action_index
        self.vel_pub.publish(cmd)

        self._wait_sim_seconds(STEP_DT)
        rclpy.spin_once(self, timeout_sec=0.05)

        reward, done = compute_reward(self.current_scan, action_index)
        self._push_frame()
        return self.get_state(), reward, done
```

Sostituisci con:
```python
    def step_action(self, action_index: int, training: bool = True):
        cmd = Twist()
        cmd.linear.x  = LINEAR_VEL
        cmd.angular.z = -0.8 + 0.16 * action_index
        self.vel_pub.publish(cmd)

        self._wait_sim_seconds(STEP_DT)
        rclpy.spin_once(self, timeout_sec=0.05)

        # Reward su scan PULITO (gradient coerente con ground-truth)
        reward, done = compute_reward(self.current_scan, action_index)

        # State su scan NOISY solo in training (perception robustness — Tobin 2017)
        if training:
            noise = np.random.normal(0, DR_NOISE_STD, LIDAR_BEAMS).astype(np.float32)
            self.current_scan = np.clip(
                self.current_scan + noise, 0.0, LIDAR_MAX_RANGE
            )

        self._push_frame()
        return self.get_state(), reward, done
```
